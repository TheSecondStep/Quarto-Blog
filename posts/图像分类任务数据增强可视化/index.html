from pathlib import Path
import torch
import matplotlib.pyplot as plt
plt.rcParams["savefig.bbox"] = 'tight'

# download helpers from https://github.com/TheSecondStep/Quarto-Blog/tree/main/posts/%E5%9B%BE%E5%83%8F%E5%88%86%E7%B1%BB%E4%BB%BB%E5%8A%A1%E6%95%B0%E6%8D%AE%E5%A2%9E%E5%BC%BA%E5%8F%AF%E8%A7%86%E5%8C%96
from helpers import plot, show_one
from torchvision.transforms import v2
from fastai.vision.all import PILImage, ToTensor

img = PILImage.create(Path('./images/rabbit.jpg').resolve())
img = PILImage.create(img.resize((512, 512)))
tfms = ToTensor()
img_t = tfms(img)
print(f"{type(img_t) = }, {img_t.dtype = }, {img_t.shape = }")

v2.Resize

tfms = v2.Resize(size=(224, 224))
out = tfms(img)
show_one(img), show_one(out)

v2.RandomCrop

tfms = v2.RandomCrop(size=(224, 224))
out = tfms(img)

plot([img, out])

v2.CenterCrop

tfms = v2.CenterCrop(size=224)
out = tfms(img)

plot([img, out])

v2.RandomResizedCrop

tfms = v2.RandomResizedCrop(size=(224, 224), scale=(0.08, 1), ratio=(3/4, 4/3))
out = tfms(img)

plot([img, out])

v2.RandomHorizontalFlip

tfms = v2.RandomHorizontalFlip(p=1)
out = tfms(img)

plot([img, out])

v2.RandomVerticalFlip

tfms = v2.RandomVerticalFlip(p=1)
out = tfms(img)

plot([img, out])

v2.Pad

tfms = v2.Pad(padding=20, padding_mode='constant')
out = tfms(img)

plot([img, out])

v2.RandomRotation

tfms = v2.RandomRotation(degrees=10)
out = tfms(img)

plot([img, out])

v2.ColorJitter

  • brightness：亮度
  • contrast：对比度
  • saturation：饱和度
  • hue：色相
# 亮度
tfms = v2.ColorJitter(brightness=0.5)
out = tfms(img)

plot([img, out])
# 对比度
tfms = v2.ColorJitter(brightness=0.5, contrast=0.5)
out = tfms(img)

plot([img, out])
tfms = v2.ColorJitter(brightness=0.5, contrast=0.5, saturation=0.5)
out = tfms(img)

plot([img, out])
tfms = v2.ColorJitter(brightness=0.5, contrast=0.5, saturation=0.5, hue=0.5)
out = tfms(img)

plot([img, out])

v2.RandAugment

tfms = v2.RandAugment()
out = tfms(img)

plot([img, out])
tfms = v2.RandAugment()
out = tfms(img)

plot([img, out])

v2.Compose

transforms = v2.Compose([
    v2.RandomResizedCrop(size=(224, 224), antialias=True),
    v2.RandomHorizontalFlip(p=0.5),
    v2.PILToTensor(),
    v2.ToDtype(torch.float32, scale=True),
    v2.Normalize(mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225]),
])
